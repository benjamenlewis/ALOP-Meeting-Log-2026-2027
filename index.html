<script>

  // ====== SETUP: paste your Apps Script Web app URL between the quotes ======

  var SCRIPT_URL = "https://script.google.com/a/macros/west40.org/s/AKfycbwmHZQn4MC6vWn-ZOPJatubPVscPpwqfbiYW3goZqGGTHYBEcNNeukz0SJkBm7c04Y_Ow/exec";

  // ==========================================================================


  var VOCAB = {

    coordinators: ["Angela","Alex","Ben","Emily","Nate","Sandra"],

    schools: ["RHS","MMS","OPRF","Hester","East Leyden","West Leyden","Scott","EPHS","Roy","Elm Middle School","Union Ridge","Westdale","Pennoyer","John Mills","Elmwood Elem","D41 Elem","District Office"],

    meetingTypes: ["Scheduled check-in","Relationship-building / intro","Problem-solving / concern","Program planning","Student-specific conversation","Family Night / event","Pop-in / unscheduled","Other"],

    roles: ["Principal","Assistant Principal","Dean","Counselor / Social Worker","Student Support Team","Superintendent / District Admin","Teacher","Parent / Guardian","Other"],

    tones: ["Very positive","Mostly positive","Neutral / mixed","Mostly negative","Very negative"],

    concernCats: ["Student behavior","Staffing / scheduling","Referral process","Caseload capacity","Communication breakdown","Resource / funding","Logistical","Student support coordination","Family / community","Other"],

    positives: ["Strong engagement / buy-in","Staff expressed appreciation","Student progress noted","Clear next steps identified","Improved relationship / trust","Program expansion opportunity","Specific advocate named","None this time"]

  };

  var entry = null;

  var REQUIRED = ["coordinator","school","meetingTypes","roles","tone","confidence","concernsRaised"];


  function $(id){ return document.getElementById(id); }

  function show(id){ ["view-gate","view-capture","view-review","view-saved"].forEach(function(v){

    $(v).classList.toggle("hidden", v !== id); }); window.scrollTo(0,0); }

  function esc(s){ var d=document.createElement("div"); d.textContent=s==null?"":String(s); return d.innerHTML; }

  function getToken(){ try { return localStorage.getItem("alop-token") || ""; } catch(e){ return ""; } }


  // ---------- API ----------

  function api(payload, onOk, onErr){

    payload.token = getToken();

    fetch(SCRIPT_URL, {

      method: "POST",

      headers: { "Content-Type": "text/plain" },

      body: JSON.stringify(payload)

    })

    .then(function(r){ return r.json(); })

    .then(function(data){

      if (data && data.ok) onOk(data);

      else onErr((data && data.error) || "Something went wrong — try again.");

    })

    .catch(function(){ onErr("Couldn't reach the server — check your connection and try again."); });

  }


  // ---------- Gate ----------

  $("gateBtn").onclick = function(){

    var t = $("gateToken").value.trim();

    var err = $("gateErr"); err.classList.add("hidden");

    if (!t){ err.textContent = "Enter the passcode."; err.classList.remove("hidden"); return; }

    try { localStorage.setItem("alop-token", t); } catch(e){}

    show("view-capture");

  };


  // ---------- Mic ----------

  var SR = window.SpeechRecognition || window.webkitSpeechRecognition;

  var rec = null, listening = false;

  function micFail(msg){

    $("micBtn").classList.add("hidden");

    var tip = $("micTip"); tip.classList.remove("hidden");

    tip.innerHTML = msg || "&#127908; The Speak button isn't supported in this browser &mdash; tap the text box, then the <b>mic key on your keyboard</b> instead.";

  }

  if (!SR) micFail();

  $("micBtn").onclick = function(){

    if (listening){ rec && rec.stop(); return; }

    try {

      rec = new SR();

      rec.continuous = true; rec.interimResults = false; rec.lang = "en-US";

      rec.onresult = function(e){

        var add = "";

        for (var i = e.resultIndex; i < e.results.length; i++)

          if (e.results[i].isFinal) add += e.results[i][0].transcript + " ";

        if (add) $("recap").value = ($("recap").value + " " + add).trim();

      };

      rec.onend = function(){ listening = false; setMicUI(); };

      rec.onerror = function(ev){

        listening = false; setMicUI();

        if (ev.error === "not-allowed" || ev.error === "service-not-allowed")

          micFail("&#127908; Microphone permission is blocked. Tap the lock icon by the address bar to allow it &mdash; or use the <b>mic key on your keyboard</b>.");

      };

      rec.start(); listening = true; setMicUI();

    } catch(e){ micFail(); }

  };

  function setMicUI(){

    $("micBtn").classList.toggle("live", listening);

    $("micLabel").textContent = listening ? "Stop" : "Speak";

  }


  // ---------- Parse ----------

  $("parseBtn").onclick = function(){

    var text = $("recap").value.trim();

    var err = $("captureErr"); err.classList.add("hidden");

    if (!text){ err.textContent = "Say or type your recap first."; err.classList.remove("hidden"); return; }

    if (listening && rec) rec.stop();

    $("parseBtn").disabled = true; $("parseBtn").textContent = "Reading your recap…";

    api({ action: "parse", transcript: text },

      function(data){

        $("parseBtn").disabled = false; $("parseBtn").textContent = "Fill in the form \u2192";

        entry = normalize(data.parsed); renderReview(); show("view-review");

      },

      function(msg){

        $("parseBtn").disabled = false; $("parseBtn").textContent = "Fill in the form \u2192";

        err.textContent = msg; err.classList.remove("hidden");

        if (/passcode/i.test(msg)) { try { localStorage.removeItem("alop-token"); } catch(e){} show("view-gate"); }

      });

  };


  function normalize(p){

    p = p || {};

    return {

      coordinator: p.coordinator || null,

      date: p.date || new Date().toISOString().slice(0,10),

      school: p.school || null,

      meetingTypes: p.meetingTypes || [],

      roles: p.roles || [],

      attendeeNames: p.attendeeNames || "",

      tone: p.tone || null,

      confidence: p.confidence || null,

      concernsRaised: (p.concernsRaised === true || p.concernsRaised === false) ? p.concernsRaised : null,

      concernCats: p.concernCats || [],

      concernDetail: p.concernDetail || "",

      commitments: p.commitments || "",

      positives: p.positives || [],

      advocateName: p.advocateName || "",

      notes: p.notes || ""

    };

  }


  // ---------- Review UI ----------

  function field(label, key, inner, needs){

    return '<div class="field"><div class="label"><span class="t">' + label + '</span>' +

      (needs ? '<span class="needs">needs you</span>' : '') + '</div>' + inner + '</div>';

  }

  function chips(key, options, multi){

    return '<div class="chips">' + options.map(function(o){

      var on = multi ? entry[key].indexOf(o) >= 0 : entry[key] === o;

      return '<button type="button" class="chip' + (on ? ' on' : '') + '" data-k="' + esc(key) +

        '" data-v="' + esc(o) + '" data-m="' + (multi?1:0) + '">' + esc(o) + '</button>';

    }).join('') + '</div>';

  }

  function isMissing(key){

    if (key === "concernsRaised") return entry.concernsRaised === null;

    var v = entry[key];

    return v === null || (Array.isArray(v) && v.length === 0);

  }

  function renderReview(){

    var h = "";

    h += field("Coordinator","coordinator", chips("coordinator", VOCAB.coordinators, false), isMissing("coordinator"));

    h += field("Date","date",'<input type="date" id="f-date" value="' + esc(entry.date) + '">', !entry.date);

    h += field("School / District","school",

      '<select id="f-school"><option value="">Choose a school…</option>' +

      VOCAB.schools.map(function(s){ return '<option' + (entry.school===s?' selected':'') + '>' + esc(s) + '</option>'; }).join('') +

      '</select>', isMissing("school"));

    h += field("Meeting type","meetingTypes", chips("meetingTypes", VOCAB.meetingTypes, true), isMissing("meetingTypes"));

    h += field("Who you met with","roles", chips("roles", VOCAB.roles, true), isMissing("roles"));

    h += field("Their names (optional)","attendeeNames",

      '<input type="text" id="f-names" placeholder="e.g. Principal Ortiz, Ms. Daly" value="' + esc(entry.attendeeNames) + '">');

    h += field("Overall tone","tone", chips("tone", VOCAB.tones, false), isMissing("tone"));

    h += field("Relationship confidence","confidence",

      '<div class="conf"><span class="lbl">At-risk</span>' +

      [1,2,3,4,5].map(function(n){ return '<button type="button" class="n' + (entry.confidence===n?' on':'') + '" data-conf="' + n + '">' + n + '</button>'; }).join('') +

      '<span class="lbl">Excellent</span></div>', isMissing("confidence"));

    h += field("Concerns raised?","concernsRaised",

      '<div class="chips">' + ["Yes","No"].map(function(o){

        var on = (entry.concernsRaised === true && o==="Yes") || (entry.concernsRaised === false && o==="No");

        return '<button type="button" class="chip' + (on?' on':'') + '" data-cr="' + o + '">' + o + '</button>';

      }).join('') + '</div>', isMissing("concernsRaised"));

    if (entry.concernsRaised === true){

      h += field("What kind?","concernCats", chips("concernCats", VOCAB.concernCats, true), entry.concernCats.length===0);

      h += field("In a sentence","concernDetail",

        '<textarea class="small" id="f-cdetail" placeholder="What happened, briefly">' + esc(entry.concernDetail) + '</textarea>');

    }

    h += field("Commitments / follow-ups","commitments",

      '<textarea class="small" id="f-commit" placeholder="What was promised, by whom, by when">' + esc(entry.commitments) + '</textarea>');

    h += field("Positive takeaways","positives", chips("positives", VOCAB.positives, true));

    if (entry.positives.indexOf("Specific advocate named") >= 0){

      h += field("Advocate's name","advocateName",

        '<input type="text" id="f-adv" placeholder="Who&#39;s championing the program?" value="' + esc(entry.advocateName) + '">');

    }

    h += field("Anything else","notes",

      '<textarea class="small" id="f-notes" placeholder="Quotes, context, things to remember">' + esc(entry.notes) + '</textarea>');

    $("fields").innerHTML = h;

    bindReview(); updateSaveBtn();

  }

  function bindReview(){

    $("fields").querySelectorAll(".chip[data-k]").forEach(function(b){

      b.onclick = function(){

        var k = b.dataset.k, v = b.dataset.v, multi = b.dataset.m === "1";

        syncTexts();

        if (multi){

          var i = entry[k].indexOf(v);

          if (i >= 0) entry[k].splice(i,1); else entry[k].push(v);

        } else entry[k] = (entry[k] === v) ? null : v;

        renderReview();

      };

    });

    $("fields").querySelectorAll("[data-conf]").forEach(function(b){

      b.onclick = function(){ syncTexts(); entry.confidence = Number(b.dataset.conf); renderReview(); };

    });

    $("fields").querySelectorAll("[data-cr]").forEach(function(b){

      b.onclick = function(){ syncTexts(); entry.concernsRaised = b.dataset.cr === "Yes"; renderReview(); };

    });

    var sch = $("f-school"); if (sch) sch.onchange = function(){ entry.school = sch.value || null; updateSaveBtn(); };

    var dt = $("f-date"); if (dt) dt.onchange = function(){ entry.date = dt.value; updateSaveBtn(); };

  }

  function syncTexts(){

    if ($("f-names")) entry.attendeeNames = $("f-names").value;

    if ($("f-cdetail")) entry.concernDetail = $("f-cdetail").value;

    if ($("f-commit")) entry.commitments = $("f-commit").value;

    if ($("f-adv")) entry.advocateName = $("f-adv").value;

    if ($("f-notes")) entry.notes = $("f-notes").value;

  }

  function requiredMissing(){

    for (var i=0;i<REQUIRED.length;i++) if (isMissing(REQUIRED[i])) return true;

    if (entry.concernsRaised === true && entry.concernCats.length === 0) return true;

    return false;

  }

  function updateSaveBtn(){

    var b = $("saveBtn"), bad = requiredMissing();

    b.disabled = bad;

    b.textContent = bad ? "Fill the highlighted fields" : "Save entry";

  }


  // ---------- Save ----------

  $("saveBtn").onclick = function(){

    syncTexts();

    if (requiredMissing()){ updateSaveBtn(); return; }

    var err = $("reviewErr"); err.classList.add("hidden");

    $("saveBtn").disabled = true; $("saveBtn").textContent = "Saving…";

    api({ action: "save", entry: entry, transcript: $("recap").value },

      function(){

        $("savedSummary").textContent = entry.school + " · " + entry.tone + " · confidence " + entry.confidence + "/5";

        show("view-saved");

        $("saveBtn").disabled = false; $("saveBtn").textContent = "Save entry";

      },

      function(msg){

        $("saveBtn").disabled = false; updateSaveBtn();

        err.textContent = msg; err.classList.remove("hidden");

      });

  };


  $("backBtn").onclick = function(){ syncTexts(); show("view-capture"); };

  $("againBtn").onclick = function(){ entry = null; $("recap").value = ""; show("view-capture"); };


  // ---------- Boot ----------

  show(getToken() ? "view-capture" : "view-gate");

</script>
